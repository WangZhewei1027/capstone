Bubble Sort Algorithm:

function bubbleSort(arr) { let n = arr.length; for (let i = 0; i < n - 1; i++) { let swapped = false; for (let j = 0; j < n - i - 1; j++) { if (arr[j] > arr[j + 1]) { // Swap elements let temp = arr[j]; arr[j] = arr[j + 1]; arr[j + 1] = temp; swapped = true; } } // If no swapping occurred, array is sorted if (!swapped) break; } return arr; }
