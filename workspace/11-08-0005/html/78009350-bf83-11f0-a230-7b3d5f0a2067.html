function binarySearch(array, target):
low = 0
high = array.length - 1
while low <= high:
mid = floor((low + high) / 2)
if array[mid] == target:
return mid // Found!
else if array[mid] < target:
low = mid + 1 // Search right half
else:
high = mid - 1 // Search left half
return -1 // Not found
