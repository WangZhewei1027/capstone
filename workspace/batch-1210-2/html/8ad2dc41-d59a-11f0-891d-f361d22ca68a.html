Red-Black Tree

Insertion

Inserting a node into the tree

            
                function insertNode(root, node) {
                    if (root === null) {
                        return node;
                    }
                    if (node.val < root.val) {
                        if (root.left === null) {
                            root.left = node;
                        } else {
                            root.left = insertNode(root.left, node);
                        }
                    } else {
                        if (root.right === null) {
                            root.right = node;
                        } else {
                            root.right = insertNode(root.right, node);
                        }
                    }
                    return root;
                }

Traversal

Traversing the tree using in-order traversal

            
                function inorderTraversal(root) {
                    const result = [];
                    function traverse(node) {
                        if (node === null) {
                            return;
                        }
                        traverse(node.left);
                        result.push(node.val);
                        traverse(node.right);
                    }
                    traverse(root);
                    return result;
                }

                function deleteNode(root, key) {
                    if (root === null) {
