K-Nearest Neighbors (KNN) Example

class KNN { constructor(data, k) { this.data = data; this.k = k; } knn(feature1, feature2) { const distances = this.data.map((point) => { const distance = Math.sqrt((point.feature1 - feature1) ** 2 + (point.feature2 - feature2) ** 2); return { point, distance }; }); const sortedDistances = distances.sort((a, b) => a.distance - b.distance); const knnResults = sortedDistances.slice(0, this.k).map((point) => point.point); return knnResults; } }
