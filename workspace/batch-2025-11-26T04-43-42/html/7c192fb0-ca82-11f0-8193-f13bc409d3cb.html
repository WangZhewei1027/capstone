Kruskal's Algorithm

Input:

        
            let edges = [
                ["A", "B", 2],
                ["B", "C", 3],
                ["C", "D", 1],
                ["D", "E", 4],
                ["E", "F", 5]
            ];

Output:

        
            let sortedEdges = [];
            while (edges.length > 0) {
                let minEdge = edges[0];
                for (let edge of edges) {
                    if (edge[2] < minEdge[2]) {
                        minEdge = edge;
                    }
                }
                sortedEdges.push(minEdge);
                edges = edges.filter(edge => edge[2] !== minEdge[2]);
            }
            console.log(sortedEdges);
