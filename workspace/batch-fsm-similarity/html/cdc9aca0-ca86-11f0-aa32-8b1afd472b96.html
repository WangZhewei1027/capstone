Heap Sort

The Heap Sort algorithm is an efficient comparison-based sorting algorithm that works by building a heap from the given array and repeatedly inserting smaller elements into the heap until the entire array is sorted.

    
      function heapSort(arr) {
        let n = arr.length;
        if (n <= 1) return arr;
        
        let heap = [];
        for (let i = 2; i < n; i++) {
          heap.push(arr[i]);
        }
        
        for (let i = n - 1; i >= 0; i--) {
          // remove the largest element from the heap
          while (!heap[0] < arr[i]) {
            heap.shift();
          }
        
          // insert the remaining elements of the heap at the correct order
          heap.push(arr[i]);
        }
        
        return heap;
      }
